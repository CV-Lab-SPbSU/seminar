\subsection{Gradient descent}

\begin{frame}
  \frametitle{Градиентный спуск}
  \begin{itemize}

    \item  Задача:

    \begin{align*}
      F : \mathbb{R}^n \rightarrow \mathbb{R} \\
      x = \argmin F(x)
    \end{align*}
   
    \item Правило приближения:

    \begin{equation*}
      x_{k+1} = x_k - \alpha_{k} \nabla F(x_k)
    \end{equation*}

    \end{itemize}
\end{frame}
